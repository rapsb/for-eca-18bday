        // DATA YANG BISA DIEDIT
        // =========================================
        
        const introTexts = [
            "Hai, Ecaa yang berusia 17 tahun...",
            "Thank you for fighting against all the struggles and your silent battles. I am so proud of you, even in the smallest things. You did great, Ecaaw!!",
            "And before you turn 18...",
            "This is probably the first and last time someone writes a website for the 17-year-old version of you.",
            "So before she disappears...",
            "I just want to say that I'm really proud of her."
        ];

        const reasons = [
            "Senyummu yang bisa mencerahkan hariku yang paling gelap sekalipun.",
            "Keteguhan hatimu dalam menghadapi setiap rintangan.",
            "Caramu tertawa yang unik dan sangat menular.",
            "Keikhlasanmu membantu orang lain tanpa pamrih.",
            "Bakat tersembunyimu yang selalu membuatku terkagum.",
            "Caramu memperhatikan hal-hal kecil yang orang lain lewatkan.",
            "Semangatmu yang tak pernah padam mengejar mimpi.",
            "Kebaikan hatimu yang tulus kepada hewan dan lingkungan.",
            "Kesabaranmu saat menghadapi situasi yang sulit.",
            "Caramu mengekspresikan perasaanmu melalui seni/kata-kata.",
            "Kehangatan yang kamu berikan kepada orang-orang di sekitarmu.",
            "Kecerdasanmu yang datang bersama dengan kerendahan hati.",
            "Semua usaha kerasmu selama ini untuk menjadi lebih baik.",
            "Kekuatanmu untuk bangkit kembali setelah setiap kegagalan.",
            "Sifat lucu dan random yang membuat hidup lebih menyenangkan.",
            "Caramu menjadi pendengar yang baik setiap saat.",
            "Potensimu yang sangat besar untuk sukses di masa depan.",
            "Dan yang terakhir: Karena kamu adalah Eca, dirimu yang sebenarnya sudah lebih dari cukup."
        ];

        const galleryData = [
            { url: "https://picsum.photos/seed/foto1/400/500", caption: "Senyum manis mu ✨" },
            { url: "https://picsum.photos/seed/foto2/400/400", caption: "Momen bahagia 🎉" },
            { url: "https://picsum.photos/seed/foto3/400/500", caption: "Jalan-jalan sore 🌆" },
            { url: "https://picsum.photos/seed/foto4/400/400", caption: "Terbaik! 🌟" },
            { url: "https://picsum.photos/seed/foto5/400/500", caption: "Kenangan tak terlupakan 💖" },
            { url: "https://picsum.photos/seed/foto6/400/400", caption: "Gadis cantik 👸" }
        ];

        const letterContent = `Dear Eca,

Selamat ulang tahun ke-18. Tak terasa waktu berlalu begitu cepat. Aku menulis ini saat kamu masih 17, namun saat kamu membaca ini, kamu sudah resmi menjadi seorang dewasa.

Ada banyak hal yang ingin kukatakan, namun seringkali tertahan. Aku bangga melihat perjalananmu. Aku melihat bagaimana kamu berjuang, bagaimana kamu bertahan, dan bagaimana kamu tetap tersenyum meski dunia terasa berat.

Di usia barumu ini, aku berharap kamu menemukan lebih banyak kebahagiaan, kedamaian, dan pencapaian. Jangan takut untuk melangkah, jangan takut untuk salah, karena dari sana kita belajar.

Teruslah bersinar seperti bintang yang selalu kukagumi. Kamu istimewa, dan jangan biarkan siapapun membuatmu merasa sebaliknya.

Seseorang yang selalu mendukungmu.`;

        // =========================================
        // LOGIC JAVASCRIPT
        // =========================================

        document.addEventListener('DOMContentLoaded', () => {
            initLoading();
        });

        /* --- 1. Loading Screen Logic --- */
        function initLoading() {
            const fill = document.getElementById('progress-fill');
            const loadingScreen = document.getElementById('loading-screen');
            
            let width = 0;
            const interval = setInterval(() => {
                if (width >= 100) {
                    clearInterval(interval);
                    setTimeout(() => {
                        loadingScreen.classList.add('fade-out');
                        setTimeout(() => {
                            loadingScreen.classList.add('hidden');
                            startIntroSequence();
                        }, 500);
                    }, 500);
                } else {
                    width += Math.random() * 5; 
                    fill.style.width = width + '%';
                }
            }, 100);
        }

        /* --- 2. Intro Sequence Logic --- */
        let introIndex = 0;
        function startIntroSequence() {
            const introSection = document.getElementById('intro-section');
            const textContainer = document.getElementById('intro-text-container');
            const btnNext = document.getElementById('btn-intro-next');

            introSection.classList.remove('hidden');
            introSection.classList.add('fade-in');

            function showText() {
                if (introIndex < introTexts.length) {
                    textContainer.innerHTML = `<p>${introTexts[introIndex]}</p>`;
                    textContainer.className = 'intro-text-box fade-in';
                    
                    introIndex++;
                    setTimeout(() => {
                        textContainer.classList.remove('fade-in');
                        textContainer.classList.add('fade-out');
                        setTimeout(() => {
                            textContainer.classList.remove('fade-out');
                            showText();
                        }, 500);
                    }, 3500); 
                } else {
                    btnNext.classList.remove('hidden');
                    btnNext.classList.add('fade-in');
                }
            }
            showText();

            btnNext.addEventListener('click', () => {
                introSection.classList.add('fade-out');
                setTimeout(() => {
                    introSection.classList.add('hidden');
                    startGame();
                }, 500);
            });
        }

        /* --- 3. Mini Game Logic --- */
        let score = 0;
        const targetScore = 18;
        let gameInterval;
        
        function startGame() {
            const gameSection = document.getElementById('game-section');
            gameSection.classList.remove('hidden');
            gameInterval = setInterval(spawnHeart, 800);
        }

        function spawnHeart() {
            if (score >= targetScore) return;

            const heart = document.createElement('div');
            heart.classList.add('heart-item');
            heart.innerHTML = '❤️';
            const leftPos = Math.random() * 85 + 5; // Fix: Jangan terlalu pinggir
            heart.style.left = leftPos + '%';
            heart.style.top = '-60px'; // Fix: Mulai dari atas layar
            const duration = Math.random() * 2 + 2; 
            heart.style.animationDuration = duration + 's';

            // Fix: Gunakan touchstart agar lebih responsif di HP
            heart.addEventListener('touchstart', (e) => {
                e.preventDefault();
                catchHeart(heart);
            }, {passive: false});

            heart.addEventListener('mousedown', (e) => {
                catchHeart(heart);
            });

            setTimeout(() => {
                if(heart.parentNode) heart.parentNode.removeChild(heart);
            }, duration * 1000);

            document.getElementById('game-section').appendChild(heart);
        }

        function catchHeart(heart) {
            if(heart.classList.contains('caught')) return;
            score++;
            heart.classList.add('caught');
            heart.innerHTML = '✨'; 
            heart.style.transform = 'scale(1.5)';
            heart.style.opacity = '0';
            document.getElementById('score-display').innerText = `${score}/${targetScore} ❤️`;
            setTimeout(() => {
                if(heart.parentNode) heart.parentNode.removeChild(heart);
            }, 300);

            if (score === targetScore) {
                clearInterval(gameInterval);
                finishGame();
            }
        }

        function finishGame() {
            const gameMsg = document.getElementById('game-msg');
            const subMsg = document.getElementById('game-sub-msg');
            gameMsg.classList.remove('hidden');
            setTimeout(() => {
                subMsg.innerHTML = "You made it.<br>Welcome to chapter 18.";
                subMsg.classList.add('fade-in');
            }, 2000);

            setTimeout(() => {
                document.getElementById('game-section').classList.add('fade-out');
                setTimeout(() => {
                    document.getElementById('game-section').classList.add('hidden');
                    showMainContent();
                }, 500);
            }, 5000);
        }

        /* --- 4. Main Content Logic --- */
        function showMainContent() {
            const mainContent = document.getElementById('main-content');
            mainContent.classList.remove('hidden');
            mainContent.classList.add('fade-in');
            
            // Fix: Aktifkan scroll body
            document.body.classList.remove('noscroll');
            
            initParticles();
            initReasons();
            initGallery();
            initLetter();
            initOutro();
            window.scrollTo(0,0);
        }

        /* --- Particles Hero --- */
        function initParticles() {
            const container = document.getElementById('particles-container');
            for(let i=0; i<30; i++) {
                const p = document.createElement('div');
                p.classList.add('particle');
                p.style.width = Math.random() * 10 + 'px';
                p.style.height = p.style.width;
                p.style.left = Math.random() * 100 + '%';
                p.style.animationDuration = Math.random() * 5 + 5 + 's';
                p.style.animationDelay = Math.random() * 5 + 's';
                container.appendChild(p);
            }
        }

        /* --- Reason Cards --- */
        function initReasons() {
            const container = document.getElementById('cards-container');
            reasons.forEach((text, index) => {
                const card = document.createElement('div');
                card.className = 'flip-card';
                card.innerHTML = `
                    <div class="flip-card-inner">
                        <div class="flip-card-front">
                            <h3>Alasan #${index + 1}</h3>
                            <p style="margin-top:10px; color:#999;">(Ketuk untuk buka)</p>
                        </div>
                        <div class="flip-card-back">
                            <p>"${text}"</p>
                        </div>
                    </div>
                `;
                card.addEventListener('click', function() {
                    this.classList.toggle('flipped');
                    if(this.classList.contains('flipped')){
                        createSparkle(this);
                    }
                });
                container.appendChild(card);
            });
        }

        function createSparkle(element) {
            const rect = element.getBoundingClientRect();
            for(let i=0; i<5; i++){
                const sparkle = document.createElement('div');
                sparkle.className = 'sparkle';
                sparkle.style.top = (Math.random() * rect.height) + 'px';
                sparkle.style.left = (Math.random() * rect.width) + 'px';
                element.appendChild(sparkle);
                setTimeout(() => sparkle.remove(), 500);
            }
        }

        /* --- Gallery --- */
        function initGallery() {
            const container = document.getElementById('gallery-container');
            const lightbox = document.getElementById('lightbox');
            const lightboxImg = document.getElementById('lightbox-img');
            const closeBtn = document.querySelector('.lightbox-close');

            galleryData.forEach(item => {
                const polaroid = document.createElement('div');
                polaroid.className = 'polaroid';
                polaroid.innerHTML = `
                    <img src="${item.url}" alt="Foto">
                    <div class="polaroid-caption">${item.caption}</div>
                `;
                polaroid.addEventListener('click', () => {
                    lightboxImg.src = item.url;
                    lightbox.classList.add('active');
                });
                container.appendChild(polaroid);
            });

            closeBtn.addEventListener('click', () => lightbox.classList.remove('active'));
            lightbox.addEventListener('click', (e) => {
                if(e.target === lightbox) lightbox.classList.remove('active');
            });
        }

        /* --- Letter Logic --- */
        function initLetter() {
            const trigger = document.getElementById('envelope-trigger');
            const overlay = document.getElementById('letter-overlay');
            const envelope = document.getElementById('envelope');
            const textElement = document.getElementById('typewriter-text');
            
            let isOpen = false;
            let typingInterval;

            function startTypewriter() {
                textElement.innerHTML = "";
                let i = 0;
                function type() {
                    if (i < letterContent.length) {
                        if (letterContent.charAt(i) === "\n") {
                            textElement.innerHTML += "<br>";
                        } else {
                            textElement.innerHTML += letterContent.charAt(i);
                        }
                        i++;
                        typingInterval = setTimeout(type, 50);
                    }
                }
                type();
            }

            function openEnvelope() {
                isOpen = true;
                envelope.classList.add('open');
                overlay.classList.add('active');
                startTypewriter();
            }

            function closeEnvelope() {
                isOpen = false;
                envelope.classList.remove('open');
                overlay.classList.remove('active');
                clearTimeout(typingInterval);
                setTimeout(() => {
                    textElement.innerHTML = "";
                }, 600);
            }

            trigger.addEventListener('click', (e) => {
                e.stopPropagation();
                if (!isOpen) {
                    openEnvelope();
                }
            });

            overlay.addEventListener('click', () => {
                if (isOpen) {
                    closeEnvelope();
                }
            });
        }

        /* --- Outro & Countdown --- */
        function initOutro() {
            const starContainer = document.getElementById('stars-container');
            for(let i=0; i<50; i++) {
                const star = document.createElement('div');
                star.className = 'star';
                star.style.left = Math.random() * 100 + '%';
                star.style.top = Math.random() * 100 + '%';
                star.style.width = Math.random() * 3 + 'px';
                star.style.height = star.style.width;
                star.style.setProperty('--duration', (Math.random() * 2 + 1) + 's');
                starContainer.appendChild(star);
            }

            const outroSection = document.getElementById('outro-section');
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(entry => {
                    if(entry.isIntersecting) {
                        launchConfetti();
                        setTimeout(() => {
                            document.getElementById('final-msg').classList.add('show');
                        }, 2000);
                        observer.unobserve(entry.target);
                    }
                });
            }, { threshold: 0.5 });
            
            observer.observe(outroSection);

            const nextBirthday = new Date(); 
            nextBirthday.setFullYear(nextBirthday.getFullYear() + 1); 
            
            setInterval(() => {
                const now = new Date();
                const diff = nextBirthday - now;
                
                const days = Math.floor(diff / (1000 * 60 * 60 * 24));
                const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((diff % (1000 * 60)) / 1000);

                document.getElementById('days').innerText = String(days).padStart(2, '0');
                document.getElementById('hours').innerText = String(hours).padStart(2, '0');
                document.getElementById('minutes').innerText = String(minutes).padStart(2, '0');
                document.getElementById('seconds').innerText = String(seconds).padStart(2, '0');
            }, 1000);
        }

        function launchConfetti() {
            const colors = ['#ffd1dc', '#fffdd0', '#ffffff', '#ff9a9e'];
            for (let i = 0; i < 100; i++) {
                const confetti = document.createElement('div');
                confetti.className = 'confetti';
                confetti.style.left = Math.random() * 100 + '%';
                confetti.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
                confetti.style.animationDuration = Math.random() * 3 + 2 + 's';
                confetti.style.opacity = Math.random();
                document.getElementById('outro-section').appendChild(confetti);
                setTimeout(() => confetti.remove(), 5000);
            }
        }
